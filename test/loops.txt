# for

for (i = 10; i; i--)
  f();

==>
Nasl(ForLoop(for,
             ForExpr(AffFunc(Identifier, Integer),
                     Identifier,
                     AffFunc(Identifier)),
             SimpleInstr(FuncCall(Identifier,
                                  ArgList))))

# for empty

for (;i;) {}

==>
Nasl(ForLoop(for,
             ForExpr(AffFunc,
                     Identifier,
                     AffFunc),
             Block))

# while

while (1)
  f();

==>
Nasl(WhileLoop(while,
               ParenExpr(Integer),
               SimpleInstr(FuncCall(Identifier,
                                    ArgList))))

# repeat

repeat f(); until a;

==>
Nasl(RepeatLoop(repeat,
                SimpleInstr(FuncCall(Identifier,
                                     ArgList)),
                until,
                Identifier))

# foreach

foreach a (b) f(a);

==>
Nasl(ForeachLoop(foreach,
                 Identifier,
                 ParenExpr(Identifier),
                 SimpleInstr(FuncCall(Identifier,
                                      ArgList(Identifier)))))
