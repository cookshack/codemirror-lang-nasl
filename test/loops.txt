# while

while (1)
  f();

==>
Nasl(WhileLoop(while,
               LoopExpr(Integer),
               SimpleInstr(FuncCall(Identifier,
                                    ArgList))))

# repeat

repeat f(); until a;

==>
Nasl(RepeatLoop(repeat,
                SimpleInstr(FuncCall(Identifier,
                                     ArgList)),
                until,
                Identifier))

# foreach

foreach a (b) f(a);

==>
Nasl(ForeachLoop(foreach,
                 Identifier,
                 LoopExpr(Identifier),
                 SimpleInstr(FuncCall(Identifier,
                                      ArgList(Identifier)))))
